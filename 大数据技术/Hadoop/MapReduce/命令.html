JARS=$(addprefix /home/hadoop/hadoop-1.2.1/,hadoop-core-1.2.1.jar lib/commons-cli-1.2.jar)

    mkdir -p build
    javac -classpath $(shell echo $(JARS) | sed "s/ /:/") \
        -d build WordCount.java
    jar -cvf wordcount.jar -C build/ .
