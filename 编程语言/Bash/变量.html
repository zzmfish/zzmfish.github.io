参数

$*  #"$1 $2 ..."  
$@  #"$1" "$2" "$3"  
$#  #参数个数  
shift [n] #弹出前面n个参数，n默认为1

内置变量

$?                  #上一个程序的退出代码  
${BASH_SOURCE[0]}   #source文件路径

变量替换

${var:-word}    #var为空则返回word
${var:=word}    #var为空则返回word，并将var的值设置为word
${var:?message} #var为空则将消息message送到标准错误输出，脚本停止运行
${var:+word}    #var被定义则返回word
