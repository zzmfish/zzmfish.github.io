软件包管理

#显示软件信息
sudo apt-cache show -a $Package  

#显示文件所属软件
apt-file search $File

#添加PPA源
sudo add-apt-repository $Package

#重新安装软件
sudo apt-get install $Package --reinstall

系统安装

#安装windows后恢复Ubuntu引导，先从Ubuntu光盘引导
mount /dev/sda10 /mnt  
grub-install --root-directory=/mnt /dev/sda  
update-grub2 #不是必须
