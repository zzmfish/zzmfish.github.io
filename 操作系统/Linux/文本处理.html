文本处理

#小写转大写
tr '[a-z]' '[A-Z]'

#计算字符个数（以“a”为例）
echo "abcabcaaaaa" | tr -dc "a" | wc -c

#显示第100行
sed '100q;d' $File

#删除空白行
sed "/^\s*$/d"

#输出匹配的组
echo "a=1 b=2 c=3" | sed "s/.*b=\([0-9]\).*/\1/" #匹配b=2，输出2

#交换两列
awk '{ print $2" "$1; }'

#在行尾添加
sed '/^CFLAGS\s*=/ s/$/ -fPIC/' -i Makefile

#在vim中打开程序输出
ps -efH | vim -

#最后一个字段内容
awk '{print $NF}'

#字段总数
awk '{print NF}'
