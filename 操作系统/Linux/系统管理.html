用户管理

#添加用户
useradd $UserName

#初始化/修改密码
passwd $UserName

#修改用户的默认shell
usermod -s /bin/bash $UserName

#添加用户到一个组
usermod -a -G $GroupName $UserName

#显示用户所在的组
groups
groups $UserName

#修改文件所有者
chown $User $File

#sudo用户配置
vi /etc/sudoers

crontab

#列出条目
crontab -l

#添加条目

rpm安装

#安装到指定位置
rpm -i --prefix=$InstallPath $RpmFile

#列出安装包的文件
rpm -qpl $RpmFile

#解压安装包
rpm2cpio $RpmFile | cpio -div
