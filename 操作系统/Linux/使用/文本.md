## 文本处理

### 选择
```
#!bash
#只转换匹配的行
sed '/^CFLAGS=/ s/$/ -fPIC/' -i Makefile

#第100行
sed '100q;d' $File
head -n 100 | tail -n 1

#最后一个字段
awk '{print $NF}'

#输出匹配的组
echo "a=1 b=2 c=3" | sed "s/.*b=\([0-9]\).*/\1/" #匹配b=2，输出2

#输出最短一行
awk '{print length($1), $1}' | sort -nk 1 | head -1 | awk '{print $2}')
```

### 变换
```
#!bash
#交换两列
awk '{ print $2" "$1; }'

#小写转大写
tr '[a-z]' '[A-Z]'
```

### 插入
```
#!bash
#文件末尾插入
sed '$a\New Line' file

#前插入
sed '/here/iNew Line'

#后插入
sed '/here/aNew Line'
```

### 删除
```
#!bash
#删除空白行
sed "/^\s*$/d"
```

### 统计
```
#!bash
#计算字符个数（以“a”为例）
echo "abcabcaaaaa" | tr -dc "a" | wc -c

#字段总数
awk '{print NF}'
```

### 其他
```
#!bash
#在vim中打开程序输出
ps -efH | vim -
```
